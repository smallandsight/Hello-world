# 古月租车小程序 - 前端接口文档

> **文档版本**：V1.0  
> **编制日期**：2026-04-05  
> **适用范围**：支付宝小程序用户端 + 商家管理后台  
> **技术栈**：uni-app (用户端) / Vue3 + Element Plus (商家端)  

---

## 一、文档说明

### 1.1 文档目的

本文档定义前端页面结构、组件规范、状态管理、路由配置等，作为前后端协作的桥梁，让后端了解前端的页面需求和数据结构。

### 1.2 两端区分

| 端 | 技术栈 | 用户群体 | 核心功能 |
|---|--------|---------|---------|
| **用户端** | uni-app + Vue3 | 租车用户 | 浏览车辆、预订下单、订单管理、支付评价 |
| **商家端** | Vue3 + Element Plus | 门店工作人员 | 车辆管理、订单处理、财务管理、数据看板 |

#### 页面数据需求

| 数据模块 | API接口 | 说明 |
|---------|---------|------|
| Banner轮播 | `GET /banners` | 活动banner、推荐车型 |
| 推荐车辆 | `GET /vehicles?recommend=true&limit=5` | 首页推荐车辆列表 |
| 附近门店 | `GET /stores/nearby` | 基于LBS的附近门店 |
| 城市列表 | `GET /cities` | 支持的城市列表 |

#### 页面Props定义

```typescript
interface IndexPageData {
  banners: Banner[];           // 轮播图列表
  recommendVehicles: Vehicle[]; // 推荐车辆
  nearbyStores: Store[];       // 附近门店
  currentCity: string;          // 当前城市
  loading: boolean;             // 加载状态
}

interface Banner {
  id: number;
  imageUrl: string;
  linkType: 'vehicle' | 'activity' | 'page';
  linkValue: string;  // 跳转参数
}

interface Vehicle {
  vehicleId: number;
  brand: string;
  model: string;
  dailyRate: number;
  image: string;
  vehicleType: number;
  storeName: string;
  rating: {
    avgScore: number;
    reviewCount: number;
  };
}
```

import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import * as userApi from '@/api/user';

export const useUserStore = defineStore('user', () => {
  // State
  const token = ref<string>('');
  const userInfo = ref<UserInfo | null>(null);
  const isLoggedIn = computed(() => !!token.value && !!userInfo.value);
  
  // Getters
  const memberLevel = computed(() => userInfo.value?.memberLevel ?? 0);
  const memberLevelText = computed(() => {
    const levels = ['普通会员', '银卡会员', '金卡会员', '钻石会员'];
    return levels[memberLevel.value] || '普通会员';
  });
  const hasLicense = computed(() => userInfo.value?.licenseStatus === 1);
  const canCreditFree = computed(() => (userInfo.value?.creditScore ?? 0) >= 650);
  
  // Actions
  const login = async (authCode: string) => {
    const res = await userApi.alipayLogin({ authCode });
    token.value = res.data.token;
    await fetchUserInfo();
    return res.data;
  };
  
  const fetchUserInfo = async () => {
    const res = await userApi.getUserInfo();
    userInfo.value = res.data;
    return res.data;
  };
  
  const logout = () => {
    token.value = '';
    userInfo.value = null;
  };
  
  const updateUserInfo = async (data: Partial<UserInfo>) => {
    await userApi.updateUserInfo(data);
    await fetchUserInfo();
  };
  
  return {
    token,
    userInfo,
    isLoggedIn,
    memberLevel,
    memberLevelText,
    hasLicense,
    canCreditFree,
    login,
    fetchUserInfo,
    logout,
    updateUserInfo
  };
}, {
  persist: {
    key: 'gy-user',
    storage: {
      getItem: (key) => uni.getStorageSync(key),
      setItem: (key, value) => uni.setStorageSync(key, value)
    }
  }
});

### 6.2 商家端Store

#### 权限状态 (stores/permission.ts)

```typescript
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import type { RouteRecordRaw } from 'vue-router';
import * as authApi from '@/api/auth';
import { generateRoutes } from '@/router/permission';

export const usePermissionStore = defineStore('permission', () => {
  // State
  const routes = ref<RouteRecordRaw[]>([]);
  const permissions = ref<string[]>([]);
  const roles = ref<string[]>([]);
  
  // Getters
  const hasPermission = computed(() => (permission: string) => {
    return permissions.value.includes(permission) || permissions.value.includes('*');
  });
  
  // Actions
  const fetchPermissions = async () => {
    const res = await authApi.getPermissions();
    permissions.value = res.data.permissions;
    roles.value = res.data.roles;
    routes.value = generateRoutes(permissions.value);
    return res.data;
  };
  
  const checkPermission = (permission: string | string[]) => {
    if (Array.isArray(permission)) {
      return permission.some(p => hasPermission.value(p));
    }
    return hasPermission.value(permission);
  };
  
  return {
    routes,
    permissions,
    roles,
    hasPermission,
    fetchPermissions,
    checkPermission
  };
});

};

// 响应拦截
const responseInterceptor = (response: UniApp.RequestSuccessCallbackResult) => {
  const { statusCode, data } = response;
  
  if (statusCode === 200) {
    const res = data as ApiResponse;
    if (res.code === 0) {
      return res;
    }
    // 业务错误
    handleBusinessError(res);
    return Promise.reject(res);
  }
  
  // HTTP错误
  if (statusCode === 401) {
    // Token过期，跳转登录
    const userStore = useUserStore();
    userStore.logout();
    uni.navigateTo({ url: '/pages/login/index' });
    return Promise.reject(new Error('登录已过期'));
  }
  
  return Promise.reject(new Error(`HTTP Error: ${statusCode}`));
};

// 统一请求方法
export const request = <T = any>(options: UniApp.RequestOptions): Promise<ApiResponse<T>> => {
  return new Promise((resolve, reject) => {
    const config = requestInterceptor(options);
    
    uni.request({
      ...config,
      url: BASE_URL + config.url,
      success: (res) => {
        responseInterceptor(res).then(resolve).catch(reject);
      },
      fail: (err) => {
        reject(new Error(err.errMsg));
      }
    });
  });
};

// GET请求
export const get = <T = any>(url: string, params?: Record<string, any>) => {
  return request<T>({
    url,
    method: 'GET',
    data: params
  });
};

// POST请求
export const post = <T = any>(url: string, data?: any) => {
  return request<T>({
    url,
    method: 'POST',
    data
  });
};

// PUT请求
export const put = <T = any>(url: string, data?: any) => {
  return request<T>({
    url,
    method: 'PUT',
    data
  });
};

// DELETE请求
export const del = <T = any>(url: string) => {
  return request<T>({
    url,
    method: 'DELETE'
  });
};
```

// 日期格式化
export const formatDate = (date: string | Date, format = 'YYYY-MM-DD HH:mm:ss'): string => {
  const d = new Date(date);
  const year = d.getFullYear();
  const month = String(d.getMonth() + 1).padStart(2, '0');
  const day = String(d.getDate()).padStart(2, '0');
  const hours = String(d.getHours()).padStart(2, '0');
  const minutes = String(d.getMinutes()).padStart(2, '0');
  const seconds = String(d.getSeconds()).padStart(2, '0');
  
  return format
    .replace('YYYY', String(year))
    .replace('MM', month)
    .replace('DD', day)
    .replace('HH', hours)
    .replace('mm', minutes)
    .replace('ss', seconds);
};

// 相对时间
export const formatRelativeTime = (date: string | Date): string => {
  const now = new Date();
  const target = new Date(date);
  const diff = target.getTime() - now.getTime();
  
  const minutes = Math.floor(Math.abs(diff) / 1000 / 60);
  const hours = Math.floor(minutes / 60);
  const days = Math.floor(hours / 24);
  
  if (diff > 0) {
    if (days > 0) return `${days}天后`;
    if (hours > 0) return `${hours}小时后`;
    return `${minutes}分钟后`;
  } else {
    if (days > 0) return `${days}天前`;
    if (hours > 0) return `${hours}小时前`;
    return `${minutes}分钟前`;
  }
};
